var options1 = { weekday: 'long', year: 'numeric', month: 'long', day: 'numeric' };
var dateTimeFormat1 = new Intl.DateTimeFormat('sr-RS', options1);

console.log(dateTimeFormat1.format(new Date(2012, 05)));
// expected output: "петак, 1. јун 2012."

var dates2 = [new Date(2012, 08), new Date(2012, 11), new Date(2012, 03)];
var options2 = { year: 'numeric', month: 'long' };
var dateTimeFormat2 = new Intl.DateTimeFormat('pt-BR', options2);

console.log(dates2.map(dateTimeFormat2.format));
// expected output: Array ["setembro de 2012", "dezembro de 2012", "abril de 2012"]
