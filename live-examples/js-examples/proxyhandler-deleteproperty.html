var handler1 = {
  deleteProperty: function(target, prop) {
    if (prop in target){
      delete target[prop];

      console.log('property removed: ' + prop);
      // expected output: "property removed: prop1"
    }
  }
}

var proxy1 = new Proxy({}, handler1);

proxy1.prop1 = 10;

console.log('prop1' in proxy1);
// expected output: true

result = delete proxy1.prop1;

console.log('prop1' in proxy1);
// expected output: false
