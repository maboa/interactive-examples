function monster(disposition) {
  this.disposition = disposition;
}

var handler = {
  construct: function(target, args) {
    console.log("monster constructor called");
    // expected output: "monster constructor called"

    return new target(...args);
  }
};

var monsterProxy = new Proxy(monster, handler);

console.log(new monsterProxy("fierce").disposition);
// expected output: "fierce"
